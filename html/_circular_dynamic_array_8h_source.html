379 void merge(int start, int mid, int end) {
380 int n1 = (mid - start + cap) % cap + 1;
381 int n2 = (end - mid + cap) % cap;
382 T* left = new T[n1];
383 T* right = new T[n2];
384
385 for (int i = 0; i < n1; i++) left[i] = data[(start + i) % cap];
386 for (int j = 0; j < n2; j++) right[j] = data[(mid + 1 + j) % cap];
387
388 int i = 0, j = 0, k = start;
389 while (i < n1 && j < n2) {
390 if (left[i] <= right[j]) {
391 data[(k++) % cap] = left[i++];
392 } else {
393 data[(k++) % cap] = right[j++];
394 }
395 }
396
397 while (i < n1) {
398 data[(k++) % cap] = left[i++];
399 }

401 while (j < n2) {
402 data[(k++) % cap] = right[j++];
403 }

460 if (k > 0 && k <= arr.size) {
461 int randomIndex = rand() % arr.size;
462
463 CircularDynamicArray<T> left, equal, right;
464 T pivot = arr[(arr.front + randomIndex) % arr.cap];
465 for (int i = 0; i < arr.size; i++) {
466 int arrayIdx = (arr.front + i) % arr.cap;
467 if (arr.data[arrayIdx] < pivot) left.addEnd(arr.data[arrayIdx]);
468 else if (arr.data[arrayIdx] == pivot) equal.addEnd(arr.data[arrayIdx]);
469 else right.addEnd(arr.data[arrayIdx]);
470 }
471
472 if (k <= left.length()) return quickSelect(left, k);
473 else if (k <= left.length() + equal.length()) return pivot;
474 else return quickSelect(right, k - left.length() - equal.length());
475 } else {
